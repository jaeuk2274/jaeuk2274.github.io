~~~java
package me.jaeuk;

import java.util.ArrayList;
import java.util.List;

/**
 * https://programmers.co.kr/learn/courses/30/lessons/42840
 * 프로그래머스 - 완전탐색 - 모의고사
 */
public class 모의고사 {
    public static void main(String[] args) {
        int[] answers = {1,3,2,4,2};

        모의고사 모의고사 = new 모의고사();
        int answer[] = 모의고사.solution(answers);

        for (int i : answer) {
            System.out.print(i + ", ");
        }
    }

    public int[] solution(int[] answers) {
        int[][] typeAns = {
                {1, 2, 3, 4, 5},
                {2, 1, 2, 3, 2, 4, 2, 5},
                {3, 3, 1, 1, 2, 2, 4, 4, 5, 5}
        };

        int typeAnsCnt[] = new int[typeAns.length]; // 케이스별 정답수 배열
        int ansIdx = 0;
        int tpIdx = 0;
        for (int ans : answers) { // 정답수만큼

            tpIdx = 0;
            for (int[] tp : typeAns) { // 케이스별 체크
                if (ans == tp[ansIdx % tp.length]) {
                    typeAnsCnt[tpIdx]++; // 케이스별 정답수++
                }
                tpIdx++;
            } // type end

            ansIdx++;
        } // answers end

        // 가장 높은 점수 체크
        int max = 0;
        for (int ansCnt : typeAnsCnt) {
            if (max < ansCnt) {
                max = ansCnt;
            }
        }

        // 가장 높은 점수로 확인 (여러명 가능/인덱스 넣어줌)
        List<Integer> returnList = new ArrayList<Integer>();
        tpIdx = 1;
        for (int ansCnt : typeAnsCnt) {
            if (max == ansCnt) {
                returnList.add(tpIdx);
            }
            tpIdx++;
        }

        // Collections.sort(returnList); // 오름차순
        // Collections.sort(returnList, Comparator.reverseOrder()); // 내림차순

        // list -> 배열로 변환
        int[] returnArray = new int[returnList.size()];
        tpIdx = 0;
        for (int rtn : returnList) {
            returnArray[tpIdx] = rtn;
            tpIdx++;
        }

        return returnArray;
    }
}
~~~

list -> array 변환시에 toList 같은 메서드를 사용하려고 했는데,    
int 가 아닌 Integer으로 해야 다시 변환하지 않았다.    
오히려 성능상 더 느리길래, 그냥 해당 방식으로 구현하였다.
